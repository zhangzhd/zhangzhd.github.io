  • 展示controller中接收的值
@RequestMapping(value = "update", method = RequestMethod.POST)
public String update(@Valid @ModelAttribute("team") Team team, RedirectAttributes redirectAttributes) {
    teamService.updateTeam(team);
    redirectAttributes.addFlashAttribute("message", "更新球队信息成功");
    return "redirect:/team";
}
  • 展示实体类
@Entity
@Table(name = "lnts_team")
public class Team extends IdEntity {
    private String teamName;
    private String registerDate;
    private String coachId;
    private String teamHome;
    private String teamCity;
    private String website;

    public String getTeamName() {
        return teamName;
    }

    public void setTeamName(String teamName) {
        this.teamName = teamName;
    }

    public String getRegisterDate() {
        return registerDate;
    }

    public void setRegisterDate(String registerDate) {
        this.registerDate = registerDate;
    }

    public String getCoachId() {
        return coachId;
    }

    public void setCoachId(String coachId) {
        this.coachId = coachId;
    }

    public String getTeamHome() {
        return teamHome;
    }

    public void setTeamHome(String teamHome) {
        this.teamHome = teamHome;
    }

    public String getTeamCity() {
        return teamCity;
    }

    public void setTeamCity(String teamCity) {
        this.teamCity = teamCity;
    }

    public String getWebsite() {
        return website;
    }

    public void setWebsite(String website) {
        this.website = website;
    }

    @Override
    public String toString() {
        return ToStringBuilder.reflectionToString(this);
    }
}
